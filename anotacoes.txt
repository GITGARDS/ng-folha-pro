import {
  addDoc,
  collection,
  deleteDoc,
  doc,
  onSnapshot,
  orderBy,
  query,
  updateDoc,
} from 'firebase/firestore';

interface iGenericsServiceI<T> {
  
  findAll(): Observable<T[]>;

  create({ data }: { data: T }): Promise<string>;

  updateById({ id, data }: { id: string; data: Partial<T> }): Promise<void>;

  deleteById({ id }: { id: string }): Promise<void>;
}

@Injectable({
  providedIn: 'root',
})
export class GenericsService<T> {
  vcollection;
  firestore = collection(db, 'tabela');
  isDelay = 0;
  orderBy = 'referencia';
  order = 'desc';

  constructor(
    @Inject(String) path: string,
    @Inject(Number) isDelay: number,
    @Inject(String) orderBy: string,
    @Inject(String) order: string,
    @Inject(collection) firestore,
  ) {
    this.vcollection = path;
    this.isDelay = isDelay;
    this.orderBy = orderBy;
    this.order = order;
  }

  findAll() {
    const q = query(this.firestore, orderBy(this.orderBy, this.order as 'asc' | 'desc'));
    return new Observable<T[]>((observer) => {
      onSnapshot(q, (snapshot) => {
        const items: T[] = snapshot.docs.map((d) => ({
          id: d.id,
          ...d.data(),
        })) as T[];
        observer.next(items);
      });
    }).pipe(delay(this.isDelay as number));
  }

  async create({ data }: { data: any }) {
    return await addDoc(this.firestore, data).then((docRef) => docRef.id);
  }

  async updateById({ id, data }: { id: string; data: Partial<T> }) {
    const docRef = doc(db, this.vcollection, id);
    await updateDoc(docRef, { ...data });
  }

  async deleteById({ id }: { id: string }) {
    const docRef = doc(db, this.vcollection, this.vcollection, id);
    await deleteDoc(docRef);
  }
}
